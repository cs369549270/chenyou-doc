# 公共属性下拉选项

}, {

    "input": "input",
    "option": []
}, {
    "label": "游戏id",
    "db_field": "property_3",
    "dtype": "number",
    "input": "input",

}]

# 公共属性筛选条件显示

根据返回的dtype 来显示

dtype 可用筛选条件（页面显示） 传参（字符串）
string 等于 =
不等于 <>
包含 like
不包含 not like
为空 is null
不为空 is not null
number 等于 =
不等于 <>
大于 >
大于等于 >=
小于 <
小于等于 <=
在...之间 between

# 开始分析

  • 传参
{
    "type":0,//分析的类型，事件分析为 0
    "event":[//事件指标列表，有对照组时，一个对照组为一个事件指标
        {
            "eventId":1,//事件唯一标识
            "indicator":1,//指标唯一标识
            "property":{//事件属性
                "outerLogic":"and",//外层逻辑，即原型中的二级逻辑
                "element":[{
                    "innerLogic":"or",//内层逻辑，即原型中的一级逻辑
                    "exprs":[{
                        "db_field":"property_1",//事件属性唯一标识
                        "filter":"=",//筛选条件
                        "targetValue":""//目标值，若是筛选条件是between，传两个值用英文逗号隔开
                    },{
                        "db_field":"property_2",
                        "filter":"=",
                        "targetValue":""
                    }]
                },{
                    "innerLogic":"or",
                    "exprs":[{
                        "db_field":"property_1",
                        "filter":"=",
                        "targetValue":""
                    }]
                }]
            }
        }
    ],
    "publicProperty":[{//公共属性列表
        "db_field":"property_1",
        "filter":"=",
        "targetValue":""
    },{
        "db_field":"property_1",
        "filter":"=",
        "targetValue":""
    }],
    "timeDimension":{//时间维度
        "timeParticle":"day",//时间粒度
        "startDate":"2020-08-20 00:00:00",//开始日期
        "endDate":"2020-08-20 00:00:00"//结束日期
    }
}

# 保存自定义指标

传参

{
    "eventId":1,//事件标识
    "indicatorName":"新建的指标名称",
    "indicatorKey":"",//新建指标的变量名
    "calculation":1,//0 表示指标运算，1 表示sql查询；当两者都不为空是，传 1
    "indicatorOp":"",//指标计算的内容
    "sql":"",//查询的sql
    "drilId":"1,2"//可下钻的指标id
}

# 获取自定义指标可用参数

{
    "option":[{
        "eventId":1,//事件标识
        "eventName":"",//事件显示名称
        "children":[{
            "indicatorId":1,//指标标识
            "indicatorName":"",//指标显示名称
            "indicatorKey":""//指标的变量名，显示时需要将名称和变量名合在一起显示，复制只复制变量名
        }]
    }]
}
